# Redirect DNS queries from Tailscale clients to dnsmasq (bypass gluetun DNS)
iptables -t nat -A PREROUTING -i tailscale0 -p udp --dport 53 -j DNAT --to-destination 172.29.0.30:53
iptables -t nat -A PREROUTING -i tailscale0 -p tcp --dport 53 -j DNAT --to-destination 172.29.0.30:53
iptables -t nat -A POSTROUTING -o eth0 -d 172.29.0.30 -p udp --dport 53 -j MASQUERADE
iptables -t nat -A POSTROUTING -o eth0 -d 172.29.0.30 -p tcp --dport 53 -j MASQUERADE
iptables -A INPUT -i tailscale0 -p udp --dport 53 -j ACCEPT
iptables -A INPUT -i tailscale0 -p tcp --dport 53 -j ACCEPT

# Allow ntfy access from Tailscale clients (push notifications)
iptables -A INPUT -i tailscale0 -p tcp --dport 80 -j ACCEPT

# Allow MagicDNS (tailscaled) to forward DNS queries to dnsmasq
iptables -A OUTPUT -o eth0 -d 172.29.0.30 -p udp --dport 53 -j ACCEPT
iptables -A OUTPUT -o eth0 -d 172.29.0.30 -p tcp --dport 53 -j ACCEPT

# Allow tailscaled to send traffic on tailscale0 (exit node responses, MagicDNS)
iptables -A OUTPUT -o tailscale0 -j ACCEPT

iptables -A FORWARD -i tailscale0 -o tun0 -j ACCEPT
iptables -A FORWARD -i tun0 -o tailscale0 -j ACCEPT
iptables -A FORWARD -i tailscale0 -o eth0 -j ACCEPT
iptables -A FORWARD -i eth0 -o tailscale0 -j ACCEPT
iptables -A FORWARD -m state --state RELATED,ESTABLISHED -j ACCEPT
iptables -t nat -A POSTROUTING -o tun0 -j MASQUERADE

# Clamp TCP MSS to fit within Tailscale MTU (1280 - 40 TCP/IP headers = 1240)
iptables -t mangle -A FORWARD -p tcp --tcp-flags SYN,RST SYN -o tailscale0 -j TCPMSS --set-mss 1240
iptables -t mangle -A FORWARD -p tcp --tcp-flags SYN,RST SYN -i tailscale0 -j TCPMSS --set-mss 1240
